h2 {
  color: #2c3e50;

.intro {
  font-weight: bold;
  font-size: 1.1em;
  color: #2980b9;
}

/* Element with class selector */
p {
  line-height: 1.6;
  margin: 10px 0;
}

/* Descendant selector */
p .highlight {
  background-color: #f1c40f;
  padding: 2px 5px;
  border-radius: 3px;
}

/* ID selector */
#special-content {
  background-color: #e74c3c;
  color: white;
  padding: 10px;
  margin: 15px 0;
  border-radius: 5px;
}

/* Element selector with pseudo-class */
ul {
  list-style-type: square;
  background-color: #f8f9fa;
  padding: 15px 40px;
  border-left: 4px solid #7f8c8d;
}

/* Class selector for specific list item */
.important {
  font-weight: bold;
  color: #c0392b;
}

/* Adjacent sibling selector */
h2 + .intro {
  margin-top: 20px;
}

  padding: 20px;
  border-radius: 5px;
  display: flex;
  justify-content: center;
  align-items: center;
  color: white;
  font-weight: bold;
}

/* Header Area */
.header {
  grid-area: header;
  background-color: #e74c3c;
}

/* Sidebar Area */
.sidebar {
  grid-area: sidebar;
  background-color: #3498db;
}

/* Main Content Area */
.main {
  grid-area: main;
  background-color: #2ecc71;
}

/* Footer Area */
.footer {
  grid-area: footer;
  background-color: #9b59b6;

/* Responsive layout for smaller screens */
@media (max-width: 600px) {

    grid-template-columns: 1fr;
    grid-template-areas:
      "header"
      "sidebar"
      "main"
      "footer";
  }
}

  border: none;
  border-radius: 4px;
  font-size: 16px;
  cursor: pointer;
  transition: all 0.3s ease;
}

.animated-button:hover {
  background-color: #2980b9;
  transform: translateY(-5px);
  box-shadow: 0 10px 20px rgba(0, 0, 0, 0.1);
}

/* Box with animation */
.animated-box {
  width: 100px;
  height: 100px;
  background-color: #e74c3c;
  animation: rotate-scale 3s ease-in-out infinite;
}

@keyframes rotate-scale {
  0% {
    transform: rotate(0deg) scale(1);
    background-color: #e74c3c;
  }
  50% {
    transform: rotate(180deg) scale(1.5);
    background-color: #2ecc71;
    border-radius: 50%;
  }
  100% {
    transform: rotate(360deg) scale(1);
    background-color: #e74c3c;
  }
}

/* Pulse element */
.pulse-element {
  padding: 15px 30px;
  background-color: #9b59b6;
  color: white;
  border-radius: 4px;
  text-align: center;
  animation: pulse 2s ease-in-out infinite;
}

@keyframes pulse {
  0% {
    transform: scale(1);
    opacity: 1;
  }
  50% {
    transform: scale(1.05);
    opacity: 0.8;
  }
  100% {
    transform: scale(1);
    opacity: 1;
  }
}
