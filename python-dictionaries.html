Python Dictionaries Implementation

Dictionary Creation & Basic Operations

// Creating dictionaries and performing basic operations

# Creating dictionaries
person = {
    'name': 'John',
    'age': 30,
    'city': 'New York'
}

# Alternative creation methods
person2 = dict(name='Alice', age=25, city='Boston')
person3 = dict([('name', 'Bob'), ('age', 35), ('city', 'Chicago')])

# Basic operations
person['job'] = 'Developer'    # Add or update key-value pair
age = person['age']            # Access a value (raises KeyError if not found)
safe_email = person.get('email', 'Not available')  # Safe access with default
del person['city']             # Remove a key-value pair
has_name = 'name' in person    # Check if key exists

print(person)
print(f"Age: {age}")
print(f"Email: {safe_email}")
print(f"Has name: {has_name}")

Dictionary Methods

// Common dictionary methods

person = {
    'name': 'John',
    'age': 30,
    'city': 'New York',
    'job': 'Developer'
}

# Dictionary methods
keys = person.keys()           # Get all keys (view object)
values = person.values()       # Get all values (view object)
items = person.items()         # Get all key-value pairs (view object)

# Copy
person_copy = person.copy()    # Shallow copy

# Pop and popitem
age = person.pop('age')        # Remove key and return value
last_item = person.popitem()   # Remove and return last inserted item (Python 3.7+)

# Update dictionary
updates = {'age': 31, 'email': 'john@example.com'}
person.update(updates)         # Update with another dictionary

# Clear
person_copy.clear()            # Remove all items

print(f"Keys: {list(keys)}")
print(f"Values: {list(values)}")
print(f"Items: {list(items)}")
print(f"Popped age: {age}")
print(f"Last item: {last_item}")
print(f"Updated person: {person}")
print(f"Empty copy: {person_copy}")

Dictionary Comprehensions

// Creating dictionaries with comprehensions

# Basic dictionary comprehension
squares = {x: x**2 for x in range(1, 6)}
# Equivalent to:
# squares = {}
# for x in range(1, 6):
#     squares[x] = x**2

# Dictionary comprehension with condition
even_squares = {x: x**2 for x in range(1, 11) if x % 2 == 0}

# Convert two lists to a dictionary
names = ['Alice', 'Bob', 'Charlie']
ages = [25, 30, 35]
name_to_age = {name: age for name, age in zip(names, ages)}

# Swapping keys and values (assuming unique values)
age_to_name = {age: name for name, age in name_to_age.items()}

print(squares)
print(even_squares)
print(name_to_age)
print(age_to_name)

Nested Dictionaries

// Working with nested dictionaries

# Creating nested dictionaries
employees = {
    'john': {
        'id': 1,
        'age': 30,
        'department': 'Engineering',
        'skills': ['Python', 'JavaScript', 'SQL']
    },
    'alice': {
        'id': 2,
        'age': 25,
        'department': 'Product',
        'skills': ['Project Management', 'UX Design']
    }
}

# Accessing nested data
john_dept = employees['john']['department']
alice_skills = employees['alice']['skills']

# Adding nested data
employees['bob'] = {
    'id': 3,
    'age': 35,
    'department': 'Marketing',
    'skills': ['Content Strategy', 'Social Media']
}

# Modifying nested data
employees['john']['skills'].append('Docker')

# Safely accessing deeply nested data
def get_nested(obj, path, default=None):
    """Access nested object with path like 'a.b.c'"""
    parts = path.split('.')
    for part in parts:
        if obj and isinstance(obj, dict) and part in obj:
            obj = obj[part]
        else:
            return default
    return obj

safe_value = get_nested(employees, 'alice.skills.0', 'Not found')

print(f"John's department: {john_dept}")
print(f"Alice's skills: {alice_skills}")
print(f"Safe access: {safe_value}")
print(f"Bob's data: {employees['bob']}")

Advanced Dictionary Operations

// More complex dictionary operations

# Merging dictionaries (Python 3.9+)
dict1 = {'a': 1, 'b': 2}
dict2 = {'b': 3, 'c': 4}

# Python 3.9+
# merged = dict1 | dict2  # {'a': 1, 'b': 3, 'c': 4}

# Python 3.5+
merged = {**dict1, **dict2}  # {'a': 1, 'b': 3, 'c': 4}

# Default dictionaries
from collections import defaultdict

# Default dictionary with default value of 0
word_counts = defaultdict(int)
for word in ['apple', 'banana', 'apple', 'orange', 'banana', 'apple']:
    word_counts[word] += 1  # No KeyError for new keys

# Default dictionary with default as list
grouped_words = defaultdict(list)
words = ['apple', 'bat', 'bar', 'atom', 'book']
for word in words:
    grouped_words[word[0]].append(word)

# OrderedDict (preserves insertion order - less relevant in Python 3.7+)
from collections import OrderedDict
ordered = OrderedDict([('a', 1), ('b', 2), ('c', 3)])

# Counter (specialized dictionary for counting)
from collections import Counter
char_count = Counter('mississippi')
most_common = char_count.most_common(2)  # Top 2 most common

print(f"Merged: {merged}")
print(f"Word counts: {dict(word_counts)}")
print(f"Grouped words: {dict(grouped_words)}")
print(f"Ordered dict: {dict(ordered)}")
print(f"Character count: {dict(char_count)}")
print(f"Most common: {most_common}")
