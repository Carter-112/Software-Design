HTML Structure and Semantics

Basic HTML Document Structure

// Write the basic structure of an HTML document

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>My Web Page</title>
  <meta name="description" content="A description of my web page">
  <link rel="stylesheet" href="styles.css">
  <script src="script.js" defer></script>
</head>
<body>
  <header>
    <h1>My Website</h1>
    <nav>
      <ul>
        <li><a href="#">Home</a></li>
        <li><a href="#">About</a></li>
        <li><a href="#">Contact</a></li>
      </ul>
    </nav>
  </header>
  
  <main>
    <h2>Welcome to My Website</h2>
    <p>This is the main content of my web page.</p>
  </main>
  
  <footer>
    <p>© 2023 My Website. All rights reserved.</p>
  </footer>
</body>
</html>

Media and Embedding

// Embedding images, videos, audio, and other content in HTML

<!-- Responsive image -->
<figure>
  <img src="sample-image.jpg" 
       alt="A scenic mountain landscape at sunset" 
       width="800" 
       height="500" 
       loading="lazy">
  <figcaption>Mountain landscape at sunset, taken by Jane Doe</figcaption>
</figure>

<!-- Picture element for responsive images -->
<picture>
  <source media="(max-width: 600px)" srcset="small-image.jpg">
  <source media="(max-width: 1200px)" srcset="medium-image.jpg">
  <img src="large-image.jpg" alt="Adaptive image based on screen size">
</picture>

<!-- SVG embedding -->
<svg width="100" height="100" viewBox="0 0 100 100">
  <circle cx="50" cy="50" r="40" stroke="black" stroke-width="2" fill="red" />
</svg>

<!-- Audio element -->
<figure>
  <figcaption>Listen to the T-Rex:</figcaption>
  <audio controls>
    <source src="trex-roar.mp3" type="audio/mpeg">
