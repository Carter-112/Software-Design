ArrayList Implementation

Class Declaration, Fields & Constructor

// The basic structure of the ArrayList class.

public class MyArrayList<E> implements Iterable<E> {
    // Default capacity when a no-arg constructor is used
    private static final int DEFAULT_CAPACITY = 10;
    
    // Array to store elements
    private Object[] elements;
    
    // Number of elements in the list
    private int size;
    
    // Constructor with initial capacity
    public MyArrayList(int initialCapacity) {
        if (initialCapacity < 0) {
            throw new IllegalArgumentException("Illegal capacity: " + initialCapacity);
        }
        this.elements = new Object[initialCapacity];
        this.size = 0;
    }
    
    // No-arg constructor with default capacity
    public MyArrayList() {
        this(DEFAULT_CAPACITY);
    }

Add Method

// Add an element to the end of the list.

public boolean add(E element) {
    ensureCapacity(size + 1);
    elements[size++] = element;
    return true;
}

private void ensureCapacity(int minCapacity) {
    // If we need more capacity, grow the array
    if (minCapacity > elements.length) {
        int newCapacity = Math.max(elements.length * 2, minCapacity);
        // Create a new array and copy elements
        elements = Arrays.copyOf(elements, newCapacity);
    }
}

Add at Index Method

// Add an element at a specific index.

public void add(int index, E element) {
    // Check if index is valid
    if (index < 0 || index > size) {
        throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
    }
    
    // Ensure we have enough space
    ensureCapacity(size + 1);
    
    // Shift elements right to make space
    System.arraycopy(elements, index, elements, index + 1, size - index);
    
    // Insert the element and increment size
    elements[index] = element;
    size++;
}

Get Method

// Get an element at a specific index.

@SuppressWarnings("unchecked")
public E get(int index) {
    if (index < 0 || index >= size) {
        throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
    }
    return (E) elements[index];
}

Remove Method

// Remove an element at a specific index.

@SuppressWarnings("unchecked")
public E remove(int index) {
    if (index < 0 || index >= size) {
        throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
    }
    
    E oldValue = (E) elements[index];
    
    // Shift elements left to fill the gap
    int numMoved = size - index - 1;
    if (numMoved > 0) {
        System.arraycopy(elements, index + 1, elements, index, numMoved);
    }
    
    // Clear the reference to the last element to help with garbage collection
    elements[--size] = null;
    
    return oldValue;
}

Remove Element Method

// Remove the first occurrence of the specified element.

public boolean remove(Object o) {
    // Find the index of the element
    for (int i = 0; i < size; i++) {
        if (o == null ? elements[i] == null : o.equals(elements[i])) {
            // Remove the element at the found index
            remove(i);
            return true;
        }
    }
    return false;
}

Set Method

// Replace an element at a specific index.

@SuppressWarnings("unchecked")
public E set(int index, E element) {
    if (index < 0 || index >= size) {
        throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
    }
    
    E oldValue = (E) elements[index];
    elements[index] = element;
    return oldValue;
}

Iterator Implementation

// Implement the Iterator interface.

@Override
public Iterator<E> iterator() {
    return new ArrayListIterator();
}

private class ArrayListIterator implements Iterator<E> {
    private int cursor = 0;
    
    @Override
    public boolean hasNext() {
        return cursor < size;
    }
    
    @SuppressWarnings("unchecked")
    @Override
    public E next() {
        if (!hasNext()) {
            throw new NoSuchElementException();
        }
        return (E) elements[cursor++];
    }
    
    @Override
    public void remove() {
        if (cursor <= 0) {
            throw new IllegalStateException();
        }
        MyArrayList.this.remove(--cursor);
    }
}

Additional Methods (size, isEmpty, clear, contains)

// Implement utility methods.

public int size() {
    return size;
}

public boolean isEmpty() {
    return size == 0;
}

public void clear() {
    // Clear references to help GC
    for (int i = 0; i < size; i++) {
        elements[i] = null;
    }
    size = 0;
}

public boolean contains(Object o) {
    return indexOf(o) >= 0;
}

public int indexOf(Object o) {
    for (int i = 0; i < size; i++) {
        if (o == null ? elements[i] == null : o.equals(elements[i])) {
            return i;
        }
    }
    return -1;
}

@Override
public String toString() {
    if (size == 0) {
        return "[]";
    }
    
    StringBuilder sb = new StringBuilder("[");
    for (int i = 0; i < size; i++) {
        sb.append(elements[i]);
        if (i < size - 1) {
            sb.append(", ");
        }
    }
    sb.append("]");
    return sb.toString();
}
