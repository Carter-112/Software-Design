/* Base styles for the image container */
.image-container {
  width: 100%;
  margin: 0 auto;
  text-align: center;
}

/* Base styles for the image */
#responsive-image {
  max-width: 100%;
  height: auto;
  display: block;
  margin: 0 auto;
  transition: all 0.3s ease;
}

.image-caption {
  margin-top: 10px;
  text-align: center;
  font-style: italic;
}

/* Desktop styles */
@media (min-width: 769px) {
  .image-container {
    width: 80%;
  }
  
  #responsive-image {
    width: 100%;
  }
  
  .image-caption::after {
    content: " (Desktop View)";
  }
}

/* Tablet styles */
@media (min-width: 481px) and (max-width: 768px) {
  .image-container {
    width: 90%;
  }
  
  #responsive-image {
    border: 5px solid #3498db;
    box-shadow: 0 5px 15px rgba(0, 0, 0, 0.3);
  }
  
  .image-caption::after {
    content: " (Tablet View)";
  }
}

/* Mobile styles */
@media (max-width: 480px) {
  .image-container {
    width: 95%;
  }
  
  #responsive-image {
    width: 80%;
    border-radius: 50%;
    border: 3px solid #e74c3c;
  }
  
  .image-caption::after {
    content: " (Mobile View)";
  }
}

/* Container for the product cards */
.card-container {
  display: flex;
  flex-wrap: wrap;
  gap: 20px;
  justify-content: space-between;
}

/* Individual product card styles */
.product-card {
  background-color: #222;
  border-radius: 8px;
  overflow: hidden;
  box-shadow: 0 4px 8px rgba(0, 0, 0, 0.2);
  transition: transform 0.3s ease, box-shadow 0.3s ease;
  display: flex;
  flex-direction: column;
}

.product-card:hover {
  transform: translateY(-5px);
  box-shadow: 0 8px 16px rgba(0, 0, 0, 0.3);
}

/* Product image placeholder */
.card-image {
  height: 150px;
  margin-bottom: 15px;
  border-radius: 0;
}

/* Card text content */
.product-card h3 {
  padding: 0 15px;
  margin-top: 0;
}

.product-card p {
  padding: 0 15px;
  margin-bottom: 20px;
  flex-grow: 1;
}

/* Card button */
.card-button {
  background-color: #3498db;
  color: white;
  border: none;
  padding: 10px 0;
  width: 100%;
  text-align: center;
  font-weight: bold;
  cursor: pointer;
  transition: background-color 0.3s ease;
  margin-top: auto;
}

.card-button:hover {
  background-color: #2980b9;
}

/* Desktop Layout (> 768px) */
@media (min-width: 769px) {
  .product-card {
    flex: 1 1 calc(25% - 15px);
    max-width: calc(25% - 15px);
  }
  
  .card-container {
    flex-wrap: nowrap;
  }
  
  .flexbox-viewport-size::after {
    content: " - Desktop Layout (Flexbox)";
  }
}

/* Tablet Layout (481px - 768px) */
@media (min-width: 481px) and (max-width: 768px) {
  .product-card {

  }
  
  .card-container {
    flex-wrap: wrap;
  }
  
  .flexbox-viewport-size::after {
    content: " - Tablet Layout (Flexbox)";
  }
}
