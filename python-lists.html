Python Lists Implementation

List Creation & Basic Operations

// Creating lists and performing basic operations

# Creating a list
numbers = [1, 2, 3, 4, 5]
fruits = ['apple', 'banana', 'orange']
mixed = [1, 'hello', True, 3.14]

# Basic operations
numbers.append(6)       # Add element to end
numbers.insert(0, 0)    # Insert at specific index
numbers.remove(3)       # Remove specific value
popped = numbers.pop()  # Remove and return last element
numbers.extend([7, 8])  # Add multiple elements

print(numbers)          # Output: [0, 1, 2, 4, 5, 7, 8]
print(len(numbers))     # Output: 7
print(3 in numbers)     # Output: False (we removed 3)

Accessing & Slicing Lists

// How to access elements and slice lists

# List for examples
letters = ['a', 'b', 'c', 'd', 'e', 'f', 'g']

# Accessing elements (0-indexed)
first = letters[0]      # 'a'
last = letters[-1]      # 'g'

# Slicing [start:end:step]
subset = letters[1:4]   # ['b', 'c', 'd'] (end index not included)
first_three = letters[:3]     # ['a', 'b', 'c']
last_three = letters[-3:]     # ['e', 'f', 'g']
every_second = letters[::2]   # ['a', 'c', 'e', 'g']
reversed_list = letters[::-1] # ['g', 'f', 'e', 'd', 'c', 'b', 'a']

print(first, last)
print(subset)
print(first_three)
print(last_three)
print(every_second)
print(reversed_list)

List Comprehensions

// Creating lists with comprehensions

# Basic list comprehension
squares = [x**2 for x in range(1, 6)]
# Equivalent to:
# squares = []
# for x in range(1, 6):
#     squares.append(x**2)

# List comprehension with condition
even_squares = [x**2 for x in range(1, 11) if x % 2 == 0]

# Nested list comprehension
matrix = [[i*j for j in range(1, 4)] for i in range(1, 4)]

# List comprehension with multiple conditions
filtered = [x for x in range(20) if x % 2 == 0 if x % 3 == 0]

print(squares)       # [1, 4, 9, 16, 25]
print(even_squares)  # [4, 16, 36, 64, 100]
print(matrix)        # [[1, 2, 3], [2, 4, 6], [3, 6, 9]]
print(filtered)      # [0, 6, 12, 18]

Sorting & List Manipulation

// How to sort and manipulate lists

numbers = [4, 2, 9, 7, 1, 3]
fruits = ['orange', 'apple', 'banana', 'kiwi']

# Sorting methods
numbers.sort()              # Sorts in-place
print(numbers)              # [1, 2, 3, 4, 7, 9]

sorted_fruits = sorted(fruits)  # Returns a new sorted list
print("Original:", fruits)      # Original: ['orange', 'apple', 'banana', 'kiwi']
print("Sorted:", sorted_fruits) # Sorted: ['apple', 'banana', 'kiwi', 'orange']

# Reverse
numbers.reverse()           # Reverses in-place
print(numbers)              # [9, 7, 4, 3, 2, 1]

# Sort with custom key
fruits.sort(key=len)        # Sort by length
print(fruits)               # ['kiwi', 'apple', 'orange', 'banana']

# Sort in reverse order
numbers.sort(reverse=True)
print(numbers)              # [9, 7, 4, 3, 2, 1]

Advanced List Operations

// More complex list operations and techniques

# List unpacking
first, second, *rest = [1, 2, 3, 4, 5]
print(first, second, rest)  # 1 2 [3, 4, 5]

# Using map and filter with lists
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, numbers))
evens = list(filter(lambda x: x % 2 == 0, numbers))

print(squared)  # [1, 4, 9, 16, 25]
print(evens)    # [2, 4]

# List as a stack (Last-In, First-Out)
stack = []
stack.append(1)    # Push
stack.append(2)
stack.append(3)
print(stack.pop()) # Pop - returns 3
print(stack)       # [1, 2]

# List as a queue (First-In, First-Out)
from collections import deque
queue = deque(["Eric", "John", "Michael"])
queue.append("Terry")
print(queue.popleft())  # Eric
print(queue)           # deque(['John', 'Michael', 'Terry'])
