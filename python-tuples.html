Python Tuples

Tuples are immutable sequences in Python, similar to lists but cannot be changed after creation. They're often used for fixed collections of items where data integrity is important.

1. Creating and Accessing Tuples

Tuples are defined using parentheses () or the tuple() constructor. Once created, elements can be accessed by index but cannot be modified.

# Creating tuples empty_tuple = () single_item_tuple = (1,) # Note the comma - needed for single item tuples mixed_tuple = (1, "hello", 3.14, True) nested_tuple = (1, (2, 3), ["a", "b"]) # Alternative creation tuple_from_list = tuple([1, 2, 3]) tuple_from_string = tuple("Python") # Accessing tuple elements print(mixed_tuple[0]) # 1 print(mixed_tuple[1:3]) # ("hello", 3.14) print(nested_tuple[1][0]) # 2

2. Tuple Methods and Operations

While tuples have fewer methods than lists due to their immutability, they support common sequence operations.

# Tuple methods numbers = (1, 2, 3, 2, 4, 2) print(numbers.count(2)) # 3 - counts occurrences of value print(numbers.index(4)) # 4 - returns first index of value # Common operations tuple1 = (1, 2, 3) tuple2 = (4, 5, 6) # Concatenation combined = tuple1 + tuple2 # (1, 2, 3, 4, 5, 6) # Repetition repeated = tuple1 * 3 # (1, 2, 3, 1, 2, 3, 1, 2, 3) # Membership testing print(2 in tuple1) # True print(5 in tuple1) # False # Length print(len(combined)) # 6

3. Tuple Packing and Unpacking

Python allows packing values into tuples and unpacking tuples into individual variables.

# Tuple packing coordinates = 3, 4 # Parentheses are optional - this creates a tuple (3, 4) # Tuple unpacking x, y = coordinates print(x) # 3 print(y) # 4 # Unpacking with * first, *middle, last = (1, 2, 3, 4, 5) print(first) # 1 print(middle) # [2, 3, 4] print(last) # 5 # Swapping values a, b = 1, 2 a, b = b, a # Swaps values using tuple packing and unpacking print(a, b) # 2 1

4. Tuples vs Lists

Tuples and lists have similar capabilities, but differ in mutability and typical use cases. Understanding their differences helps choose the right data structure.

# Performance comparison import sys import timeit # Size comparison list_ex = [1, 2, 3, 4, 5] tuple_ex = (1, 2, 3, 4, 5) print(f"List size: {sys.getsizeof(list_ex)} bytes") print(f"Tuple size: {sys.getsizeof(tuple_ex)} bytes") # Creation speed list_creation = timeit.timeit(stmt="[1, 2, 3, 4, 5]", number=1000000) tuple_creation = timeit.timeit(stmt="(1, 2, 3, 4, 5)", number=1000000) print(f"List creation: {list_creation:.6f} seconds") print(f"Tuple creation: {tuple_creation:.6f} seconds") # Access speed list_access = timeit.timeit(stmt="x[2]", setup="x = [1, 2, 3, 4, 5]", number=1000000) tuple_access = timeit.timeit(stmt="x[2]", setup="x = (1, 2, 3, 4, 5)", number=1000000) print(f"List access: {list_access:.6f} seconds") print(f"Tuple access: {tuple_access:.6f} seconds")

5. Named Tuples

Named tuples are a subclass of tuples that have field names and expanded functionality while retaining tuple's immutability.

from collections import namedtuple # Creating a named tuple type Person = namedtuple('Person', ['name', 'age', 'job']) # Creating instances alice = Person(name='Alice', age=30, job='Engineer') bob = Person('Bob', 25, 'Designer') # Positional arguments work too # Accessing by name or position print(alice.name) # 'Alice' print(bob[1]) # 25 # Converting to dictionary alice_dict = alice._asdict() print(alice_dict) # {'name': 'Alice', 'age': 30, 'job': 'Engineer'} # Creating new instance with replaced values alice_promoted = alice._replace(job='Senior Engineer') print(alice_promoted) # Person(name='Alice', age=30, job='Senior Engineer')

Practice: Use named tuples to represent structured data

Solution: Named Tuples

from collections import namedtuple # Create a Point namedtuple Point = namedtuple('Point', ['x', 'y', 'z']) # Create instances p1 = Point(1, 2, 3) p2 = Point(x=4, y=5, z=6) print(f"Point 1: {p1}") print(f"Point 2 coordinates: x={p2.x}, y={p2.y}, z={p2.z}") # Calculate distance between points import math distance = math.sqrt(sum((a - b) ** 2 for a, b in zip(p1, p2))) print(f"Distance between points: {distance:.2f}") # Create a Rectangle namedtuple using Points Rectangle = namedtuple('Rectangle', ['top_left', 'bottom_right']) # Create a rectangle using Point instances rect = Rectangle(Point(0, 10, 0), Point(10, 0, 0)) print(f"Rectangle: {rect}") # Calculate width and height width = rect.bottom_right.x - rect.top_left.x height = rect.top_left.y - rect.bottom_right.y print(f"Rectangle dimensions: {width}x{height}") # Create a modified rectangle that's moved by 5 units in x direction moved_rect = rect._replace( top_left=rect.top_left._replace(x=rect.top_left.x + 5), bottom_right=rect.bottom_right._replace(x=rect.bottom_right.x + 5) ) print(f"Moved rectangle: {moved_rect}") # Demonstrate _fields and _asdict() print(f"Point fields: {Point._fields}") print(f"Rectangle as dict: {rect._asdict()}") # Create a namedtuple from a dictionary point_data = {'x': 100, 'y': 200, 'z': 300} p3 = Point(**point_data) print(f"Point from dictionary: {p3}") # Create a subclass with default values class PointWithDefaults(namedtuple('PointWithDefaults', ['x', 'y', 'z'])): def __new__(cls, x=0, y=0, z=0): return super().__new__(cls, x, y, z) def magnitude(self): return math.sqrt(self.x**2 + self.y**2 + self.z**2) # Use the subclass origin = PointWithDefaults() p4 = PointWithDefaults(3, 4, 5) print(f"Default point: {origin}") print(f"Point p4 magnitude: {p4.magnitude():.2f}")
